# String Operations

## split & iterate over a string with IFS
###~
export IFS=";"
sentence="one;two;three"
for word in $sentence; do
  echo "$word"
done
~###

## split & iterate over a string without IFS
###~
#local str rest
rest=$STR
while [ -n "$rest" ] ; do
   str=${rest%%;*}  # Everything up to the first ';'
   # Trim up to the first ';' -- and handle final case, too.
   [ "$rest" = "${rest/;/}" ] && rest= || rest=${rest#*;}
   echo "+ \"$str\""
done
~###

# File operations using AWK

## awk syntax
### awk '{action}' your_file_name.txt
### awk '/regex pattern/{action}' your_file_name.txt

## awk: print all contents using
### awk '{print $0}' information.txt

## awk: print line number count using NR
### awk '{print NR,$0}' information.txt 

## awk: print 1st column
### awk '{print $1}' information.txt

## awk: print 2nd column
